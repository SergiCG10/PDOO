def manageHit(receivedAttack)
	defense = defensiveEnergy()
			
	if defense < receivedAttack
		gotWounded()
		incConsecutiveHits()
	else
		resetHits()
	end
	
	if (@consecutiveHits == @@HITS2LOSE) || dead()
		lose = true
	else
		lose = false
	end
	
	return lose
end

def move(direction, validMoves)
	
	size = validMoves.size
	contained = false
	i = 0
	while ( i < size && !contained ) 
		if validMoves[i] == direction
			contained = true
		end
		i += 1
	end
	if (size > 0 && !contained)
		firstElement = validMoves[0]
		move = firstElement
	else
		move = direction
	end
	return move
end

	    w = newWeapon
		receiveWeapon(w)
	end

def to_s
	info ="\n\t"+@name+"\n"
	info +="\n\tIntelligence: " + @intelligence.to_s
	info +="\n\tStregth: " + @strength.to_s
	info +="\n\tHealth: " + @health.to_s
	info +="\n\tPosition: (" + @row.to_s + ", " + @col.to_s + ")"
	info +="\n\n\tWeapons:\n"
	weapons.each do |weapon| 
		info+="\t" + weapon.to_s
		info+="\n"
	end
	info +="\n\n\tShields:\n"
	shields.each do |shield| 
		info+="\t" + shield.to_s
		info+="\n"
	end
	return info
